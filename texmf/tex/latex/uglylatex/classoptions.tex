% Document class language mode.
%
% Changing the language mode will change the guide words for chart titles
% (graphs, tables); article structure words, such as the `table of
% contents', `references', etc.; and the guide words in the theorem
% environment, such as `theorem', `lemma', etc.
%
\DeclareStringOption[cn]{lang}
\DeclareVoidOption{cn}{\kvsetkeys{UGLY}{lang=cn}}
\DeclareVoidOption{en}{\kvsetkeys{UGLY}{lang=en}}
